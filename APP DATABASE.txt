import sqlite3
import os

DB_PATH = os.path.join("database", "autodj.db")


def init_db():
    os.makedirs("database", exist_ok=True)

    conn = sqlite3.connect(DB_PATH)
    cursor = conn.cursor()

    cursor.execute("""
    CREATE TABLE IF NOT EXISTS media (
        id INTEGER PRIMARY KEY AUTOINCREMENT,
        title TEXT,
        path TEXT,
        type TEXT
    )
    """)

    conn.commit()
    conn.close()


def insert_media(title, path, media_type):
    conn = sqlite3.connect(DB_PATH)
    cursor = conn.cursor()

    cursor.execute("""
    INSERT INTO media (title, path, type)
    VALUES (?, ?, ?)
    """, (title, path, media_type))

    conn.commit()
    conn.close()


def get_stats():
    conn = sqlite3.connect(DB_PATH)
    cursor = conn.cursor()

    cursor.execute("SELECT COUNT(*) FROM media WHERE type='audio'")
    audio = cursor.fetchone()[0]

    cursor.execute("SELECT COUNT(*) FROM media WHERE type='video'")
    video = cursor.fetchone()[0]

    conn.close()

    return audio, video
